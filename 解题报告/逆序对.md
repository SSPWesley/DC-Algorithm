2021-8 CSG-TEST Online Judge 1049：逆序对 - 解题报告（钟承文）
# 内容 #
## 题目大意 ##
对于一个序列a，如果有ai > aj且i < j，则称ai, aj为一逆序对。现给定一个序列，求出序列中逆序对的数量（序列中可能存在重复数字）。
## 解题思路 ##
该题使用双重for进行判断那会超时，因此可以使用归并排序，降低其时间复杂度。由于归并排序每次会将左右两个区间按从小到大进行排序并合并为一个大区间，因此在合并的同时，就统计左右区间一共产生了多少逆序对即可。
## 代码 ##
```C++
#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
#include<cstring>
using namespace std;
#define M 500005
int a[M], t[M], n;
long long ans = 0;
void msort(int l, int r)
{
	if (l == r) return;
	int mid = l + r >> 1;
	msort(l, mid);	
	msort(mid + 1, r);

	int i = l, j = mid + 1, k = l;
	while (i <= mid && j <= r)
		if (a[i] <= a[j]) t[k++] = a[i++];
		else t[k++] = a[j++], ans += mid - i + 1;//统计逆序对
	while (i <= mid) t[k++] = a[i++];
	while (j <= r)   t[k++] = a[j++];
	for (int i = l; i <= r; i++) a[i] = t[i];
}
int main()
{
	cin >> n;
	for (int i = 1; i <= n; i++) cin >> a[i];
	msort(1, n);
	cout << ans;
	return 0;
}
```
